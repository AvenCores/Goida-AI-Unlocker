pyinstaller main.py --onefile --noconsole --icon=icon.ico --name="Goida_AI_Unlocker_Windows" --version-file=version.txt --add-data "icon.ico;." --add-data "app_info.json;." --add-data "icons;icons" --add-data "app;app"

Linux:
pyinstaller main.py --onefile --noconsole --icon=icon.ico --name="Goida_AI_Unlocker_Linux" --add-data "icon.ico:." --add-data "app_info.json:." --add-data "icons:icons" --add-data "app:app"

macOS (single-arch .app bundle):
pyinstaller main.py --onedir --windowed --icon=icon.icns --name="Goida_AI_Unlocker_macOS" --add-data "icon.icns:." --add-data "icon.ico:." --add-data "app_info.json:." --add-data "icons:icons" --add-data "app:app"

macOS Universal (Intel + Apple Silicon):
1. Build x86_64 .app on an Intel Mac or GitHub Actions runner (macos-13).
2. Build arm64 .app on an Apple Silicon Mac or GitHub Actions runner (macos-latest).
3. Merge the two bundles:
   python3 .github/scripts/merge_macos_apps.py \
     --x86_64 Goida_AI_Unlocker_macOS_x86_64.app \
     --arm64 Goida_AI_Unlocker_macOS_arm64.app \
     --output Goida_AI_Unlocker_macOS_Universal.app
4. Package:
   ditto -c -k --keepParent Goida_AI_Unlocker_macOS_Universal.app Goida_AI_Unlocker_macOS_Universal.zip

Build commands for step 1 and 2:
pyinstaller main.py --onedir --windowed --icon=icon.icns --name="Goida_AI_Unlocker_macOS_x86_64" --add-data "icon.icns:." --add-data "icon.ico:." --add-data "app_info.json:." --add-data "icons:icons" --add-data "app:app"
pyinstaller main.py --onedir --windowed --icon=icon.icns --name="Goida_AI_Unlocker_macOS_arm64" --add-data "icon.icns:." --add-data "icon.ico:." --add-data "app_info.json:." --add-data "icons:icons" --add-data "app:app"
